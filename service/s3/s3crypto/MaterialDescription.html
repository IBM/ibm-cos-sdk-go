func (md MaterialDescription) Clone() (clone MaterialDescription) {
	if md == nil {
		return nil
	}
	clone = make(MaterialDescription, len(md))
	for k, v := range md {
		clone[k] = copyPtrString(v)
	}
	return clone
}
